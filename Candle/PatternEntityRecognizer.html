def recognize(text, tokenizer = nil)
  entities = []
  
  @patterns.each do |pattern|
    regex = pattern.is_a?(Regexp) ? pattern : Regexp.new(pattern)
    
    text.scan(regex) do |match|
      match_text = $&
      match_start = $~.offset(0)[0]
      match_end = $~.offset(0)[1]
      
      entities << {
        "text" => match_text,
        "label" => @entity_type,
        "start" => match_start,
        "end" => match_end,
        "confidence" => 1.0,
        "source" => "pattern"
      }
    end
  end
  
  entities
end
