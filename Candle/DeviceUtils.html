def self.best_device
  # Try devices in order of preference
  begin
    # Try Metal first (for Mac users)
    Device.metal
  rescue
    # :nocov:
    begin
      # Try CUDA next (for NVIDIA GPU users)
      Device.cuda
    rescue
      # Fall back to CPU
      Device.cpu
    end
    # :nocov:
  end
end
